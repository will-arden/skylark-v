init:
	nop
    nop
	addi x2, x0, 10
    addi x3, x0, 15
    beq x2, x3, equal
    bne x2, x2, ne
    blt x2, x0, lt
    bge x3, x2, ge
    jal x1, terminate
    
equal:
   	addi x5, x0, 14
   	sw x5, 0(x0)
    jal x1, terminate
ne:
   	addi x5, x0, 30
   	sw x5, 0(x0)
    jal x1, terminate
lt:
	addi x5, x0, 255
    sw x5, 0(x0)
    jal x1, terminate
ge:
	addi x5, x0, 1
    sw x5, 0(x0)
    jal x1, terminate
    
terminate:
	jal x1, terminate
